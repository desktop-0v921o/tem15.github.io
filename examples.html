Приклади коду

Базові приклади популярних мов програмування

HTML

<!DOCTYPE html>
<html>
<head>
    <title>Приклад</title>
</head>
<body>
    <h1>Привіт, світ!</h1>
</body>
</html>

CSS

.card {
    background: #fff;
    padding: 20px;
    border-radius: 8px;

JavaScript

function greet(name) {
    return `Привіт, ${name}!`;
}
console.log(greet("Світ"));

Java

public class Hello {
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}

Python

def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n-1)
print(factorial(5))

C#

using System;

class Program {
    static void Main() {
        Console.WriteLine("Hello, World!");
    }
}

C++

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, World!" << endl;
    return 0;
}

Lua

function greet(name)
    return "Привіт, " .. name .. "!"
end

print(greet("Світ"))
